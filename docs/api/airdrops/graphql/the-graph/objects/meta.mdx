export const Details = ({ dataOpen, dataClose, children, startOpen = false }) => {
  const [open, setOpen] = useState(startOpen);
  return (
    <details
      {...(open ? { open: true } : {})}
      className="details"
      style={{ border: "none", boxShadow: "none", background: "var(--ifm-background-color)" }}
    >
      <summary
        onClick={(e) => {
          e.preventDefault();
          setOpen((open) => !open);
        }}
        style={{ listStyle: "none" }}
      >
        {open ? dataOpen : dataClose}
      </summary>
      {open && children}
    </details>
  );
};

The type for the top-level \_meta field

```graphql
type _Meta_ {
  block: _Block_!
  deployment: String!
  hasIndexingErrors: Boolean!
}
```

### Fields

#### [<code style={{ fontWeight: 'normal' }}>\_Meta\_.<b>block</b></code>](#)<Bullet />[<code style={{ fontWeight: 'normal' }}><b>\_Block\_!</b></code>](/docs/api/airdrops/graphql/the-graph/objects/block.mdx) <Badge class="badge badge--secondary badge--non_null" text="non-null"/> <Badge class="badge badge--secondary " text="object"/>

Information about a specific subgraph block. The hash of the block will be null if the \_meta field has a block
constraint that asks for a block number. It will be filled if the \_meta field has no block constraint and therefore
asks for the latest block
