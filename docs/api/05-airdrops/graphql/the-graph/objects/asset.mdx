  id: String!
  address: Bytes!
  chainId: BigInt!
  decimals: BigInt!

  symbol: String!
  campaigns(
    skip: Int = 0
    first: Int = 100
    orderBy: Campaign_orderBy
    orderDirection: OrderDirection

}
