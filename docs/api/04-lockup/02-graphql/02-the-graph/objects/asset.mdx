```graphql
type Asset {
  id: String!
  address: Bytes!
  chainId: BigInt!
  decimals: BigInt!
  name: String!
  symbol: String!
  streams(
    skip: Int = 0
    first: Int = 100
    orderBy: Stream_orderBy
    orderDirection: OrderDirection
    where: Stream_filter
  ): [Stream!]!
}
```
